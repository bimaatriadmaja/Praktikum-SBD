CREATE TABLE Mahasiswa (
NIM_Mahasiswa VARCHAR(45) PRIMARY KEY,
Nama_Mahasiswa VARCHAR(255) NOT NULL,
Alamat_Mahasiswa VARCHAR(255) NOT NULL
);

CREATE TABLE Dosen (
NIK_Dosen VARCHAR(45) PRIMARY KEY,
Nama_Dosen VARCHAR(255) NOT NULL,
Alamat_Dosen VARCHAR(255) NOT NULL
);

CREATE TABLE Mata_Kuliah (
Kode_MK VARCHAR(45) PRIMARY KEY,
NIK_DosenFK VARCHAR(45) REFERENCES Dosen(NIK_Dosen)
ON DELETE CASCADE ON UPDATE CASCADE,
Nama_MK VARCHAR(255) NOT NULL,
SKS INTEGER DEFAULT 0 NOT NULL
);

CREATE TABLE Ruang_Kuliah (
Kode_Ruang VARCHAR(45) PRIMARY KEY,
Kode_MKFK VARCHAR(45) REFERENCES Mata_Kuliah(Kode_MK)
ON DELETE SET NULL ON UPDATE CASCADE,
Nama_Ruang VARCHAR(45) NOT NULL,
Tempat_Ruang VARCHAR(45) NOT NULL
);

CREATE TABLE Mahasiswa_has_Mata_Kuliah (
NIM_MahasiswaFK VARCHAR(45) REFERENCES Mahasiswa(NIM_Mahasiswa)
ON DELETE CASCADE ON UPDATE CASCADE,
Kode_MKFK VARCHAR(45) REFERENCES Mata_Kuliah(Kode_MK)
ON DELETE CASCADE ON UPDATE CASCADE,
PRIMARY KEY(NIM_MahasiswaFK, Kode_MKFK)
);
